\subsection{Respuestas}

\begin{enumerate}
    \item 
    \[
        I_L(t) = \begin{cases}
            -1 & t < 0 \\
            2.4 - 3.4 e^{-t} & t \geq 0
        \end{cases}
    \]
    \item \ldots
    \item 
    \[
        V_C(t) = \begin{cases}
            \frac{20}{3} & t < 0 \\
            \frac{20}{3} e^{-25t} & t \geq 0
        \end{cases}
    \]
    \[
        I_C(t) = \begin{cases}
            0 & t < 0 \\
            \frac{e^{-25t}}{3000} & t \geq 0
        \end{cases}
    \]
\end{enumerate}
