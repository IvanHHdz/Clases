% ====== Comandos personalizados para imágenes ======

% Una imagen centrada
% Uso: \imagen{ruta}{Texto del pie de figura}{200px}
\newcommand{\imagen}[3]{
    \begin{figure}[ht!]
        \centering
        \includegraphics[width={#3}]{#1}
        \caption{#2}
    \end{figure}
}

% Una imagen a la par del texto
% Uso: \imagenwrap{ruta}{texto del pie de figura}{200px}{r}
\newcommand{\imagenwrap}[4]{
    \begin{wrapfigure}{#4}{#3} % l - r
        \centering
        \includegraphics[width=\linewidth]{#1}
        \caption{#2}
    \end{wrapfigure}
}

% Una imagen al par de otra
% Uso:\imagenparinline{img1}{img2}{texto}{0.45}
\newcommand{\imagenparinline}[4]{%
    \noindent\makebox[\linewidth][c]{%
        \begin{minipage}{#4\linewidth}
            \centering
            \begin{minipage}{0.49\linewidth}
                \centering
                \includegraphics[width=\linewidth]{#1}
            \end{minipage}\hfill
            \begin{minipage}{0.49\linewidth}
                \centering
                \includegraphics[width=\linewidth]{#2}
            \end{minipage}
            \vspace{2mm}
            {\small #3}
        \end{minipage}%
    }%
}
